import React from 'react'
import Draggable from 'react-draggable'
 
 
/**
  * Note component.
  */
class Note extends React.Component {
 
    componentWillMount() {
        this.style = {
            right: this.randomBetween(0, window.innerWidth - 150, 'px'),
            top: this.randomBetween(0, window.innerHeight - 150, 'px')
        }
    }
 
    randomBetween(x, y, s) {
        return (x + Math.ceil(Math.random() * (y - x))) + s
    }
 
    /**
    * to render the notes
    */
    renderDisplay() {
        return (
            <div className="note" style={this.style}>
                <p>{this.props.children}</p>
                <span>
                    <button id="edit" onClick={() => {this.props.handleEditClicked(this.props.id)}}>edit</button>
                    <button id="delete" onClick={() => {this.props.handleDeleteClicked(this.props.id)}}>X</button>
                </span>
            </div>
        )
    }
 
    /**
    * to render the text area in the note being edited.
    */
    renderForm() {
        return (
            <div className="note" style={this.style}>
                <textarea id="newText" ref="newText" defaultValue={this.props.children}>
                </textarea>
                <button onClick={() => { this.props.handleSaveClicked({ id: this.props.id, val: this.refs.newText.value })}}>
                    Save
                </button>
            </div>
        )
    }
 
    /**
    * rendering the component.
    * this.props.editing.includes(this.props.id) -- does not work in ie or edge
    */
    render() {
        return (
            <Draggable>
                {(this.props.editing.indexOf(this.props.id) !== -1)
                    ? this.renderForm() : this.renderDisplay()}
            </Draggable>
        )
    }
}
 
 
export default Note;
