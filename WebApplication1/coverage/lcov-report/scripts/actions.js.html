﻿import constants from './constants'
import { notesService } from './service/NotesService'
 
/**
 * action for creating a new note
 */
export const newNote = () => (dispatch) => {
    console.log('in new note action');
    notesService(constants.ADD_NOTE, {id: 0, val: ""}).then(res => {
        console.log(res.data);
        dispatch(addNote(res.data))
    }, (err) => {
        dispatch(addError(err))
        console.log('error:' + err)
    });
}
 
/**
 * action for updating notes array with new note.
 * @param {any} note
 * new note to be added
 */
export function addNote(note) {
    console.log('in add note action');
    return {
        type: constants.ADD_NOTE,
        payload: note
    }
}
 
/**
 * action for removing a new note
 */
export const removeNote = (id) => (dispatch) => {
    console.log('in remove note action');
    notesService(constants.DELETE_NOTE, id).then(res => {
        console.log(res.data);
        dispatch(deleteNote(id))
    }, (err) => {
        dispatch(addError(err))
        console.log('error:' + err)
    });
}
 
/**
 * action to delete a note
 * @param {any} id
 * id of the note to be deleted
 */
export function deleteNote(id) {
    console.log('in delete note action');
    return {
        type: constants.DELETE_NOTE,
        payload: id
    }
}
 
 
/**
 * action for creating a new note
 */
export const saveNote = (note) => (dispatch) => {
    console.log('in save note action');
    notesService(constants.UPDATE_NOTE, note).then(res => {
        dispatch(updateNote(note))
        dispatch({
            type: constants.EDIT_DONE,
            payload: note.id
        })
    }, (err) => {
        dispatch(addError(err))
        console.log('error:' + err)
    });
}

export function updateNote(note) {
    console.log('in update note action');
    return {
        type: constants.UPDATE_NOTE,
        payload: note
    }
}
