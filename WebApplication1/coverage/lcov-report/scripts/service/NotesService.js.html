ï»¿import constants from '.././constants'
import axios from 'axios'
 
 
/**
 * method to communicate with the notes api.
 * @param {any} service
 * wat service to call
 * @param {any} payload
 * payload for the service
 */
export function notesService(service, payload) {
    console.log("requested service:" + service)
 
    switch (service) {
    case constants.GET_NOTES:
        console.log('fetching all notes');
        
        return axios.get(constants.NOTES_API_URL);
 
    case constants.ADD_NOTE:
        console.log('adding a note');
        return axios.post(constants.NOTES_API_URL, payload);
 
    case constants.UPDATE_NOTE:
        console.log('updating a note');
        return axios.put(constants.NOTES_API_URL + '/' + payload.id, payload);
 
    case constants.DELETE_NOTE:
        console.log('deleting a note');
        return axios.delete(constants.NOTES_API_URL + '/' + payload);
 
    default:
        return null;
    }
}
